#include <iostream>
#include <vector>
using namespace std;

int mountain(int n, vector<int>mount){
    int k=0,m;
    if(n%2==0){
        m=n/2-1;
    }else{
        m=n/2;
    }
    for(int i=m;i>0;i--){
        if(mount[i]<=mount[i-1]){
            mount[i-1]--;
            k++;
        }
    }
    cout<<k;
    for(int i=0,j=n-1;i<=m;i++,j--){
        if(mount[i]!=mount[j] && i!=j){
            k++;
        }
    }
    return k;
}
int main() {
    int n;
    cin>>n;
    vector<int>mount(n,0);
    for(int i =0;i<n;i++){
        cin>>mount[i];
    }
    cout<<mountain(n,mount);
    return 0;
}
