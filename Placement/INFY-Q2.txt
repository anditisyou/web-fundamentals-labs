#include <iostream>
#include <vector>
using namespace std;

int fight(int n, int m, vector<int> h, vector<int> v ){
    int i,j;
    for(i = n-1,j = m-1; i>=0 ; i--){
        h[j]-=v[i];
        if(h[j]<0){
            break;
        }
        if(m!=1){
            j--;
        }
    }
    return i+1;
}
int main() {
    int n,m,hm;
     
    cin>>n;
    cin>>m;
    cin>>hm;
    
    vector <int> v(n,0);
    vector <int> h(m,hm);
    
    for(int i=0;i<n;i++){
        cin>>v[i];
    }
    cout<<"Output  :  "<<fight(n,m,h,v);
    return 0;
}
